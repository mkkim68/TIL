```python
import sys  
s = str(sys.stdin.readline())  
s = s[:-1]  
n = 0  
  
while len(s) != 0:  
    if s.find("c=") != -1:  
        n += 1  
        s = s[2:]  
        print(len(s))  
    elif s.find("c-") != -1:  
        n += 1  
        s = s[2:]  
    elif s.find("dz=") != -1:  
        n += 2  
        s = s[3:]  
    elif s.find("d-") != -1:  
        n += 1  
        s = s[2:]  
    elif s.find("lj") != -1:  
        n += 2  
        s = s[2:]  
    elif s.find("nj") != -1:  
        n += 2  
        s = s[2:]  
    elif s.find("s=") != -1:  
        n += 1  
        s = s[2:]  
    elif s.find("z=") != -1:  
        n += 1  
        s = s[2:]  
    else:  
        s = s[1:]  
  
print(n)
```
