```python
import sys  
input = sys.stdin.readline  
  
C, R = map(int, input().split())  
K = int(input())  
  
num = R  
flag = 1  # 1: 증가, 0: 감소  
ansC, ansR = 0, 0  
while K > 0:  
    if K > C * R:  
        print(0)  
        break  
    if K <= num:  
        if flag == 1:  
            ansR += K  
        else:  
            ansR -= K  
    elif num < K <= 2 * num:  
        if flag == 1:  
            ansR = num  
        else:  
            ansR -= num  
    K -= 2 * num  
    num -= 1  
    if flag == 1:  
        flag = 0  
    else:  
        flag = 1  
else:  
    print(ansC, ansR)
```
