```python
import sys  
input = sys.stdin.readline  
  
T = int(input())  
for _ in range(T):  
    A, B = map(int, input().split())  
    A2 = A  
    B2 = B  
    a = []  
    b = []  
    i = j = 2  
    while A > 1:  
        if A % i != 0:  
            i += 1  
            continue  
        else:  
            A //= i  
            a.append(i)  
    while B > 1:  
        if B % j != 0:  
            j += 1  
            continue  
        else:  
            B //= j  
            b.append(j)  
    max_num = 1  
    for n in a:  
        for m in b:  
            if n == m:  
                max_num *= n  
    print(a)  
    print(b)  
    print(max_num)  
    print(A2, B2)  
    print(A2 * B2 // max_num)
```
